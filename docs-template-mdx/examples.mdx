# Examples

## Basic Captcha

<Playground>
  {() => {
    const [captchaCode, setCaptcha] = React.useState();
    return (
      <div>
        <ClientCaptcha captchaCode={setCaptcha} />

      </div>
    );

}}

</Playground>

## Captcha With Input and Validation

<Playground>
  {() => {
    const [captchaCode, setCaptcha] = React.useState();
    const [captchaInput, setCaptchaInput] = React.useState();
    const validateCaptcha = () => {
        if(captchaCode === captchaInput)
            alert("Valid")
        else
            alert("Invalid")
    }
    return (
      <div>
        <ClientCaptcha captchaCode={setCaptcha} />
        <input type="text" value={captchaInput} onChange={e => setCaptchaInput(e.target.value)} />
        <button onClick={validateCaptcha}>Validate</button>
      </div>
    );

}}

</Playground>
